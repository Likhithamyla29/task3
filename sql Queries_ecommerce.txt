--Top 5 Products by Total Sales
SELECT Product_Name,
       SUM(Total_Price) AS Total_Sales
FROM ecommerce
GROUP BY Product_Name
ORDER BY Total_Sales DESC
LIMIT 5;

--Total Revenue by Category
SELECT Category,
       SUM(Total_Price) AS Total_Revenue
FROM ecommerce
GROUP BY Category;

--Number of Orders in Each Region
SELECT Region,
       COUNT(*) AS Total_Orders
FROM ecommerce
GROUP BY Region;

--Average Customer Age by Gender
SELECT Gender,
       AVG(Age) AS Avg_Age
FROM ecommerce
GROUP BY Gender;

--Top 10 Customers by Spending
SELECT Customer_ID,
       SUM(Total_Price) AS Total_Spent
FROM ecommerce
GROUP BY Customer_ID
ORDER BY Total_Spent DESC
LIMIT 10;

--Total Quantity Sold by Category
SELECT Category,
       SUM(Quantity) AS Total_Quantity
FROM ecommerce
GROUP BY Category;

--Average Shipping Fee by Shipping Status
SELECT Shipping_Status,
       AVG(Shipping_Fee) AS Avg_Shipping_Fee
FROM ecommerce
GROUP BY Shipping_Status;

--Month-wise Total Revenue
SELECT MONTH(Order_Date) AS Month_No,
       SUM(Total_Price) AS Total_Revenue
FROM ecommerce
GROUP BY MONTH(Order_Date)
ORDER BY Month_No;

--Products with Sales Greater Than 10000
SELECT Product_Name,
       SUM(Total_Price) AS Total_Sales
FROM ecommerce
GROUP BY Product_Name
HAVING SUM(Total_Price) > 10000;

--Region with Highest Revenue
SELECT Region,
       SUM(Total_Price) AS Total_Revenue
FROM ecommerce
GROUP BY Region
ORDER BY Total_Revenue DESC
LIMIT 1;
